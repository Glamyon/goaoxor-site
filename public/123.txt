    (function(){
      const defaultLang = localStorage.getItem('preferredLang') || 'en';
      const root = document.documentElement;

      function applyLang(lang){
        root.lang = (lang === 'zh') ? 'zh-CN' : 'en';
        document.querySelectorAll('.translatable').forEach(el => {
          // prefer data-zh / data-en; for complex content this file already contains translated blocks
          const key = (lang === 'zh') ? 'zh' : 'en';
          // dataset uses camelCase for hyphen keys; data attributes: data-en / data-zh -> accessible as dataset.en / dataset.zh
          if(el.dataset && (el.dataset[key] !== undefined)){
            // choose innerHTML when the data contains HTML-ish content (contains '<' char)
            const val = el.dataset[key];
            if(val && val.includes('<')) el.innerHTML = val;
            else el.textContent = val;
          }
        });

        // update lang buttons active state
        document.querySelectorAll('.lang-btn').forEach(btn=>{
          const is = btn.dataset.lang === lang || (btn.getAttribute('data-lang') === lang);
          if(is){btn.classList.add('active'); btn.setAttribute('aria-pressed','true')}
          else {btn.classList.remove('active'); btn.setAttribute('aria-pressed','false')}
        });

        localStorage.setItem('preferredLang', lang);
      }

      // setup language button listeners
      document.querySelectorAll('.lang-btn').forEach(btn=>{
        btn.addEventListener('click', e=>{
          const lang = btn.dataset.lang || btn.getAttribute('data-lang') || btn.textContent.trim().toLowerCase().startsWith('zh') ? 'zh' : 'en';
          applyLang(lang);
        });
      });

      // initialize with default
      applyLang(defaultLang);
    })();

      const closer = document.getElementById('modal-close');

          e.preventDefault();
          modal.classList.add('open');
          modal.setAttribute('aria-hidden','false');
          // lock scroll
          document.body.style.overflow = 'hidden';
          // focus
          closer && closer.focus();
        });
      });

      function close(){
        modal.classList.remove('open');
        modal.setAttribute('aria-hidden','true');
        document.body.style.overflow = '';
      }

      closer && closer.addEventListener('click', close);
      modal.addEventListener('click', e=>{
        if(e.target === modal) close();
      });

      document.addEventListener('keydown', e=>{
        if(e.key === 'Escape' && modal.classList.contains('open')) close();
      });
    })();

    /****** Count-up animation for stats ******/
    (function(){
      function animateCount(el, to){
        // skip if contains non-digit like '$' or '+'
        const hasSymbol = /[^0-9]/.test(String(to));
        // if dataset has number or text like "500+" we handle gracefully
        const raw = el.getAttribute('data-count') || el.textContent.replace(/[^0-9]/g,'');
        const target = Number(raw) || 0;
        if(!target) return;
        const dur = 1200;
        const start = performance.now();
        const initial = 0;
        function step(now){
          const t = Math.min(1,(now-start)/dur);
          const v = Math.floor(initial + (target - initial)*easeOutCubic(t));
          el.textContent = (hasSymbol ? el.textContent : v.toLocaleString());
          if(t < 1) requestAnimationFrame(step);
          else { el.textContent = el.getAttribute('data-count') || target.toLocaleString(); }
        }
        requestAnimationFrame(step);
      }
      function easeOutCubic(t){return 1 - Math.pow(1-t,3)}

      // start when visible
      const stats = document.querySelectorAll('.stat .num');
      const observer = new IntersectionObserver((entries, obs)=>{
        entries.forEach(entry=>{
          if(entry.isIntersecting){
            animateCount(entry.target);
            obs.unobserve(entry.target);
          }
        });
      }, {threshold:0.45});
      stats.forEach(s=>observer.observe(s));
    })();

    /****** Scroll reveal using IntersectionObserver ******/
    (function(){
      const items = document.querySelectorAll('section, .process-step, .service, .testimonial, .success-story');
      const io = new IntersectionObserver((entries, obs)=>{
        entries.forEach(entry=>{
          if(entry.isIntersecting){
            entry.target.style.transform = 'none';
            entry.target.style.opacity = '1';
            entry.target.style.transition = 'opacity .6s ease, transform .6s ease';
            obs.unobserve(entry.target);
          }
        });
      }, {threshold:0.12});
      items.forEach(it=>{
        it.style.opacity = '0';
        it.style.transform = 'translateY(12px)';
        io.observe(it);
      });
    })();
